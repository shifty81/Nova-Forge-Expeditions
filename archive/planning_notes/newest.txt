1️⃣ EXACT MODE CONTROLLER — CODE SKELETON (AUTHORITATIVE)
This is the single choke point for FPS ⇄ Flight ⇄ Fleet.

Core Types
enum class PlayerMode {

Mode Interface (no cross-talk allowed)
class IPlayerMode {
public:
    virtual ~IPlayerMode() = default;

    virtual void OnEnter() = 0;
    virtual void OnExit() = 0;
    virtual void Tick(float dt) = 0;

    // hard ownership
    virtual void BindInput() = 0;
    virtual void UnbindInput() = 0;

    virtual void ActivateUI() = 0;
    virtual void DeactivateUI() = 0;

    virtual void ActivateCamera() = 0;
    virtual void DeactivateCamera() = 0;
};
PlayerModeController (this owns everything)
class PlayerModeController {
public:
    void RegisterMode(PlayerMode mode, std::unique_ptr<IPlayerMode> impl);
    void RequestMode(PlayerMode target);
    void Tick(float dt);

    PlayerMode GetCurrent() const { return current; }

private:
    PlayerMode current = PlayerMode::FPS;
    PlayerMode pending = PlayerMode::FPS;

    std::unordered_map<PlayerMode, std::unique_ptr<IPlayerMode>> modes;

    void Transition();
};
Transition Logic (NO SHORTCUTS)
void PlayerModeController::RequestMode(PlayerMode target) {
    if (target == current) return;
    pending = target;
}

void PlayerModeController::Tick(float dt) {
    if (pending != current) {
        Transition();
    }
    modes[current]->Tick(dt);
}

void PlayerModeController::Transition() {
    auto& from = modes[current];
    auto& to   = modes[pending];

    from->UnbindInput();
    from->DeactivateUI();
    from->DeactivateCamera();
    from->OnExit();

    // optional: fade / time-slow hook here

    to->OnEnter();
    to->ActivateCamera();
    to->ActivateUI();
    to->BindInput();

    current = pending;
}
Rule:
If something feels wrong → fix the transition, not the mode.

Combat State Update System
class PlayerCombatStateSystem {
public:
    void Tick(Entity player, float dt) {
        auto& combat = ecs.Get<PlayerCombatStateComponent>(player);

        switch (combat.state) {
            case PlayerCombatState::EXPLORING:
                if (EnemyDetected(player))
                    Transition(player, PlayerCombatState::ALERT);
                break;

            case PlayerCombatState::ALERT:
                if (PlayerFired(player) || TookDamage(player))
                    Transition(player, PlayerCombatState::ENGAGED);
                break;

            case PlayerCombatState::ENGAGED:
                if (combat.suitHealth < 40.f)
                    Transition(player, PlayerCombatState::WOUNDED);
                if (NoEnemiesNearby())
                    Transition(player, PlayerCombatState::EXPLORING);
                break;

            case PlayerCombatState::WOUNDED:
                ApplyMovementPenalty(player);
                if (combat.suitHealth <= 0.f)
                    Transition(player, PlayerCombatState::DOWNED);
                break;

            case PlayerCombatState::BREACHED:
                ApplyZeroGModifiers(player);
                break;

            case PlayerCombatState::DOWNED:
                DisableInput(player);
                break;
        }
    }

private:
    void Transition(Entity player, PlayerCombatState next) {
        auto& combat = ecs.Get<PlayerCombatStateComponent>(player);
        combat.state = next;
        EventBus::Emit(PlayerCombatStateChanged{player, next});
    }
};

struct LowPolyMesh {
    std::vector<Vec3> positions;
    std::vector<Vec3> normals;
    std::vector<uint32_t> indices;
};
LowPolyMesh MakeFlatBox(Vec3 size) {
    LowPolyMesh m;

    // 6 faces, 4 verts each (no sharing!)
    for (Face f : BoxFaces(size)) {
        Vec3 n = f.normal;
        for (Vec3 p : f.vertices) {
            m.positions.push_back(p);
            m.normals.push_back(n);
        }
        AppendQuadIndices(m.indices);
    }
    return m;
}
One-Mutation Rule
void ApplyMutation(LowPolyMesh& m) {
    switch (RandomInt(0,2)) {
        case 0: ScaleAxis(m, Axis::X, 0.8f); break;
        case 1: InsetFace(m, RandomFace(), 0.4f); break;
        case 2: RemoveFace(m, RandomFace()); break;
    }
}

void ApplyDamage(LowPolyMesh& m, const DamageEvent& d) {
    for (Face& f : m.faces) {
        if (Distance(f.center, d.position) < d.radius) {
            if (Random() < d.severity) {

        }
    }

void ApplyBurn(Room& r, const DamageEvent& d) {
    for (Surface& s : r.surfaces) {
        float dist = Distance(s.center, d.position);
        if (dist < d.radius) {
            s.color = Lerp(s.color, Color::Black, 0.4f);
        }
    }
}

// Base abstraction for any editor tool
class ITool
{
public:
    virtual ~ITool() = default;
    virtual void Activate() = 0;
    virtual void Deactivate() = 0;
    virtual void Update(float deltaTime) = 0;
};

// Core Tooling Layer
class ToolingLayer
{
private:
    std::vector<std::unique_ptr<ITool>> activeTools;

public:
    void RegisterTool(std::unique_ptr<ITool> tool);
    void UnregisterTool(ITool* tool);
    void UpdateTools(float deltaTime);
};
Key Points:

All editor tools (e.g., StarSystemEditor, ShipModuleEditor) implement ITool.

ToolingLayer manages activation/deactivation and updates.

Keeps editor code isolated from game runtime logic.

3️⃣ EditorCommandBus Architecture
// Commands are decoupled from their execution context
struct ICommand
{
    virtual ~ICommand() = default;
    virtual void Execute() = 0;
};

// Concrete commands
class CreateStarSystemCommand : public ICommand
{
public:
    void Execute() override;
};

// Bus decouples sender/receiver
class EditorCommandBus
{
private:
    std::queue<std::unique_ptr<ICommand>> commandQueue;

public:

    void ProcessCommands();
};
Highlights:

Editor/Tooling systems post commands; bus handles execution.

class RuntimeBootstrap
{
